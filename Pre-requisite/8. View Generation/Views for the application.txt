create or replace view public.company_locations_view as
SELECT
  c.company_id,
  c.name AS company_name,
  c.short_name AS company_short_name,
  COALESCE(
    array_agg(DISTINCT ct.city ORDER BY ct.city)
    FILTER (WHERE ct.city IS NOT NULL),
    ARRAY[]::text[]
  ) AS office_cities,
  COALESCE(
    array_agg(DISTINCT co.country ORDER BY co.country)
    FILTER (WHERE co.country IS NOT NULL),
    ARRAY[]::text[]
  ) AS operating_countries
FROM companies c
LEFT JOIN company_office_locations_map olm
  ON olm.company_id = c.company_id
LEFT JOIN cities ct
  ON ct.city_id = olm.city_id
LEFT JOIN company_operating_countries_map ocm
  ON ocm.company_id = c.company_id
LEFT JOIN countries co
  ON co.countries_id = ocm.countries_id
GROUP BY c.company_id, c.name, c.short_name;


create or replace view public.company_skills_view as
SELECT
  c.company_id,
  c.name AS company_name,
  c.short_name AS company_short_name,
  sm.skill_set_id,
  sm.skill_set_name,
  sm.short_name AS skill_short_name,
  sm.skill_set_description,
  csl.required_level,
  pl.proficiency_level_id,
  pl.proficiency_name,
  pl.proficiency_code,
  pl.proficiency_description
FROM company_skill_levels csl
JOIN companies c
  ON c.company_id = csl.company_id
JOIN skill_set_master sm
  ON sm.skill_set_id = csl.skill_set_id
JOIN proficiency_levels pl
  ON pl.proficiency_level_id = csl.required_proficiency_level_id;


create or replace view public.company_student_fit_view as
WITH skill_summary AS (
  SELECT
    company_skill_levels.company_id,
    round(avg(company_skill_levels.required_level), 2) AS avg_required_level,
    max(company_skill_levels.required_level) AS max_required_level,
    count(*) AS total_required_skills
  FROM company_skill_levels
  GROUP BY company_skill_levels.company_id
),
learning AS (
  SELECT
    company_learning_culture.company_talent_growth_id,
    string_agg(
      DISTINCT NULLIF(TRIM(BOTH FROM company_learning_culture.learning_culture), ''),
      ' | '
    ) AS learning_culture
  FROM company_learning_culture
  GROUP BY company_learning_culture.company_talent_growth_id
),
mentorship AS (
  SELECT
    company_mentorship_availability.company_talent_growth_id,
    string_agg(
      DISTINCT NULLIF(TRIM(BOTH FROM company_mentorship_availability.mentorship_availability), ''),
      ' | '
    ) AS mentorship_availability
  FROM company_mentorship_availability
  GROUP BY company_mentorship_availability.company_talent_growth_id
),
exit_ops AS (
  SELECT
    company_exit_opportunities.company_talent_growth_id,
    string_agg(
      DISTINCT NULLIF(TRIM(BOTH FROM company_exit_opportunities.exit_opportunities), ''),
      ' | '
    ) AS exit_opportunities
  FROM company_exit_opportunities
  GROUP BY company_exit_opportunities.company_talent_growth_id
),
global_exp AS (
  SELECT
    company_global_exposure.company_talent_growth_id,
    string_agg(
      DISTINCT NULLIF(TRIM(BOTH FROM company_global_exposure.global_exposure), ''),
      ' | '
    ) AS global_exposure
  FROM company_global_exposure
  GROUP BY company_global_exposure.company_talent_growth_id
),
fit_inputs AS (
  SELECT
    c.company_id,
    c.name,
    c.short_name,
    c.category,
    ss.avg_required_level,
    ss.max_required_level,
    ss.total_required_skills,
    tg.onboarding_quality,
    tg.internal_mobility,
    tg.role_clarity,
    tg.early_ownership,
    tg.skill_relevance,
    tg.exposure_quality,
    tg.brand_value,
    tg.external_recognition,
    cu.manager_quality,
    cu.psychological_safety,
    cu.burnout_risk,
    cu.avg_retention_tenure,
    f.annual_revenue,
    f.valuation,
    f.yoy_growth_rate,
    f.profitability_status,
    l.learning_culture,
    m.mentorship_availability,
    eo.exit_opportunities,
    ge.global_exposure,
    br.glassdoor_rating,
    br.google_rating,
    br.indeed_rating
  FROM companies c
  LEFT JOIN skill_summary ss
    ON ss.company_id = c.company_id
  LEFT JOIN company_talent_growth tg
    ON tg.company_id = c.company_id
  LEFT JOIN company_culture cu
    ON cu.company_id = c.company_id
  LEFT JOIN company_financials f
    ON f.company_id = c.company_id
  LEFT JOIN company_brand_reputation br
    ON br.company_id = c.company_id
  LEFT JOIN learning l
    ON l.company_talent_growth_id = tg.company_talent_growth_id
  LEFT JOIN mentorship m
    ON m.company_talent_growth_id = tg.company_talent_growth_id
  LEFT JOIN exit_ops eo
    ON eo.company_talent_growth_id = tg.company_talent_growth_id
  LEFT JOIN global_exp ge
    ON ge.company_talent_growth_id = tg.company_talent_growth_id
),
score_parts AS (
  SELECT
    fi.*,

    CASE
      WHEN fi.avg_required_level IS NULL THEN 60
      WHEN fi.avg_required_level <= 3 THEN 90
      WHEN fi.avg_required_level <= 4 THEN 80
      WHEN fi.avg_required_level <= 5 THEN 68
      WHEN fi.avg_required_level <= 6 THEN 55
      ELSE 42
    END AS skill_accessibility_score,

    CASE
      WHEN lower(
        COALESCE(fi.onboarding_quality, '') || ' ' ||
        COALESCE(fi.role_clarity, '') || ' ' ||
        COALESCE(fi.early_ownership, '')
      ) ~ 'excellent|industry-leading|strong|high' THEN 85
      WHEN lower(
        COALESCE(fi.onboarding_quality, '') || ' ' ||
        COALESCE(fi.role_clarity, '') || ' ' ||
        COALESCE(fi.early_ownership, '')
      ) ~ 'moderate|medium' THEN 65
      WHEN lower(
        COALESCE(fi.onboarding_quality, '') || ' ' ||
        COALESCE(fi.role_clarity, '') || ' ' ||
        COALESCE(fi.early_ownership, '')
      ) ~ 'low|poor' THEN 40
      ELSE 60
    END AS fresher_friendliness_score,

    CASE
      WHEN lower(
        COALESCE(fi.learning_culture, '') || ' ' ||
        COALESCE(fi.mentorship_availability, '') || ' ' ||
        COALESCE(fi.exposure_quality, '')
      ) ~ 'excellent|strong|high|very high' THEN 85
      WHEN lower(
        COALESCE(fi.learning_culture, '') || ' ' ||
        COALESCE(fi.mentorship_availability, '') || ' ' ||
        COALESCE(fi.exposure_quality, '')
      ) ~ 'moderate|medium' THEN 65
      WHEN lower(
        COALESCE(fi.learning_culture, '') || ' ' ||
        COALESCE(fi.mentorship_availability, '') || ' ' ||
        COALESCE(fi.exposure_quality, '')
      ) ~ 'low|limited' THEN 40
      ELSE 60
    END AS learning_mentorship_score,

    CASE
      WHEN lower(
        COALESCE(fi.internal_mobility, '') || ' ' ||
        COALESCE(fi.exit_opportunities, '') || ' ' ||
        COALESCE(fi.global_exposure, '') || ' ' ||
        COALESCE(fi.brand_value, '')
      ) ~ 'excellent|strong|high|very high' THEN 85
      WHEN lower(
        COALESCE(fi.internal_mobility, '') || ' ' ||
        COALESCE(fi.exit_opportunities, '') || ' ' ||
        COALESCE(fi.global_exposure, '') || ' ' ||
        COALESCE(fi.brand_value, '')
      ) ~ 'moderate|medium' THEN 65
      WHEN lower(
        COALESCE(fi.internal_mobility, '') || ' ' ||
        COALESCE(fi.exit_opportunities, '') || ' ' ||
        COALESCE(fi.global_exposure, '') || ' ' ||
        COALESCE(fi.brand_value, '')
      ) ~ 'low|limited' THEN 40
      ELSE 60
    END AS career_growth_score,

    CASE
      WHEN lower(COALESCE(fi.burnout_risk, '')) LIKE '%high%'
        AND lower(COALESCE(fi.burnout_risk, '')) LIKE '%moderate%' THEN 55
      WHEN lower(COALESCE(fi.burnout_risk, '')) LIKE '%high%' THEN 35
      WHEN lower(COALESCE(fi.burnout_risk, '')) LIKE '%low%' THEN 90
      WHEN lower(COALESCE(fi.burnout_risk, '')) LIKE '%medium%'
        OR lower(COALESCE(fi.burnout_risk, '')) LIKE '%moderate%' THEN 65
      ELSE 60
    END AS work_life_balance_score,

    CASE
      WHEN lower(
        COALESCE(fi.brand_value, '') || ' ' ||
        COALESCE(fi.external_recognition, '')
      ) ~ 'excellent|strong|high|very high' THEN 85
      WHEN lower(
        COALESCE(fi.brand_value, '') || ' ' ||
        COALESCE(fi.external_recognition, '')
      ) ~ 'moderate|medium' THEN 65
      WHEN lower(
        COALESCE(fi.brand_value, '') || ' ' ||
        COALESCE(fi.external_recognition, '')
      ) ~ 'low|limited' THEN 40
      ELSE 60
    END AS brand_value_score,

    CASE
      WHEN lower(COALESCE(fi.profitability_status, '')) LIKE '%profitable%'
        AND lower(COALESCE(fi.yoy_growth_rate, '')) !~ 'negative|decline|loss' THEN 85
      WHEN lower(COALESCE(fi.profitability_status, '')) LIKE '%profitable%' THEN 75
      WHEN lower(COALESCE(fi.profitability_status, '')) LIKE '%loss%'
        OR lower(COALESCE(fi.profitability_status, '')) LIKE '%unprofitable%' THEN 45
      ELSE 60
    END AS company_stability_score
  FROM fit_inputs fi
)
SELECT
  company_id,
  name,
  short_name,
  category,
  avg_required_level,
  max_required_level,
  total_required_skills,
  skill_accessibility_score,
  fresher_friendliness_score,
  learning_mentorship_score,
  career_growth_score,
  work_life_balance_score,
  brand_value_score,
  company_stability_score,
  round(
    (
      skill_accessibility_score::numeric * 0.25 +
      fresher_friendliness_score::numeric * 0.20 +
      learning_mentorship_score::numeric * 0.15 +
      career_growth_score::numeric * 0.15 +
      work_life_balance_score::numeric * 0.10 +
      brand_value_score::numeric * 0.10 +
      company_stability_score::numeric * 0.05
    )
  )::integer AS student_fit_score,
  array_remove(
    ARRAY[
      CASE WHEN skill_accessibility_score >= 75 THEN 'Skill requirements are student-accessible' END,
      CASE WHEN fresher_friendliness_score >= 75 THEN 'Good fresher-friendly signals' END,
      CASE WHEN learning_mentorship_score >= 75 THEN 'Strong learning and mentorship potential' END,
      CASE WHEN career_growth_score >= 75 THEN 'Good career growth potential' END,
      CASE WHEN work_life_balance_score >= 75 THEN 'Healthy work-life balance signals' END,
      CASE WHEN brand_value_score >= 75 THEN 'Strong brand value' END,
      CASE WHEN company_stability_score >= 75 THEN 'Stable company profile' END
    ],
    NULL
  ) AS positive_signals,
  array_remove(
    ARRAY[
      CASE WHEN skill_accessibility_score < 60 THEN 'Higher skill expectations' END,
      CASE WHEN fresher_friendliness_score < 60 THEN 'May need stronger preparation as a fresher' END,
      CASE WHEN learning_mentorship_score < 60 THEN 'Learning or mentorship signals are limited' END,
      CASE WHEN work_life_balance_score < 60 THEN 'Possible work-life balance concern' END,
      CASE WHEN company_stability_score < 60 THEN 'Company stability needs closer review' END
    ],
    NULL
  ) AS caution_signals
FROM score_parts;


create or replace view public.company_search_view as
WITH logo_details AS (
  SELECT
    company_logo.company_id,
    min(NULLIF(TRIM(BOTH FROM company_logo.logo_url), '')) AS logo_url
  FROM company_logo
  GROUP BY company_logo.company_id
),
tech_rating AS (
  SELECT
    company_tech_adoption_rating.company_technologies_id,
    string_agg(
      DISTINCT NULLIF(TRIM(BOTH FROM company_tech_adoption_rating.tech_adoption_rating), ''),
      ' | '
    ) AS tech_adoption_rating
  FROM company_tech_adoption_rating
  GROUP BY company_tech_adoption_rating.company_technologies_id
),
top_skills_ranked AS (
  SELECT
    csv.company_id,
    csv.skill_set_name,
    csv.required_level,
    row_number() OVER (
      PARTITION BY csv.company_id
      ORDER BY csv.required_level DESC, csv.skill_set_name
    ) AS rn
  FROM company_skills_view csv
),
top_skills AS (
  SELECT
    top_skills_ranked.company_id,
    array_agg(
      top_skills_ranked.skill_set_name
      ORDER BY top_skills_ranked.required_level DESC, top_skills_ranked.skill_set_name
    ) FILTER (WHERE top_skills_ranked.rn <= 5) AS top_skills
  FROM top_skills_ranked
  GROUP BY top_skills_ranked.company_id
)
SELECT
  c.company_id,
  c.name,
  c.short_name,
  ld.logo_url,
  c.category,
  c.nature_of_company,
  c.incorporation_year,
  c.employee_size,
  c.office_count,
  c.website_url,
  c.linkedin_url,
  left(c.overview_text, 260) AS overview_preview,
  f.profitability_status,
  f.yoy_growth_rate,
  f.valuation,
  tg.skill_relevance,
  tg.brand_value,
  tg.exposure_quality,
  tg.onboarding_quality,
  cu.burnout_risk,
  cu.manager_quality,
  tech.ai_ml_adoption_level,
  tr.tech_adoption_rating,
  br.glassdoor_rating,
  br.google_rating,
  br.website_rating,
  COALESCE(ts.top_skills, ARRAY[]::text[]::varchar[]) AS top_skills,
  COALESCE(loc.office_cities, ARRAY[]::text[]) AS office_cities,
  COALESCE(loc.operating_countries, ARRAY[]::text[]) AS operating_countries,
  fit.student_fit_score
FROM companies c
LEFT JOIN logo_details ld
  ON ld.company_id = c.company_id
LEFT JOIN company_financials f
  ON f.company_id = c.company_id
LEFT JOIN company_talent_growth tg
  ON tg.company_id = c.company_id
LEFT JOIN company_culture cu
  ON cu.company_id = c.company_id
LEFT JOIN company_technologies tech
  ON tech.company_id = c.company_id
LEFT JOIN tech_rating tr
  ON tr.company_technologies_id = tech.company_technologies_id
LEFT JOIN company_brand_reputation br
  ON br.company_id = c.company_id
LEFT JOIN top_skills ts
  ON ts.company_id = c.company_id
LEFT JOIN company_locations_view loc
  ON loc.company_id = c.company_id
LEFT JOIN company_student_fit_view fit
  ON fit.company_id = c.company_id;


create or replace view public.company_profile_view as
WITH logo_details AS (
  SELECT
    company_logo.company_id,
    min(NULLIF(TRIM(BOTH FROM company_logo.logo_url), '')) AS logo_url
  FROM company_logo
  GROUP BY company_logo.company_id
),
business_details AS (
  SELECT
    b.company_id,
    string_agg(DISTINCT NULLIF(TRIM(BOTH FROM fs.focus_sectors), ''), ' | ')
      FILTER (WHERE NULLIF(TRIM(BOTH FROM fs.focus_sectors), '') IS NOT NULL) AS focus_sectors,
    string_agg(DISTINCT NULLIF(TRIM(BOTH FROM od.offerings_description), ''), ' | ')
      FILTER (WHERE NULLIF(TRIM(BOTH FROM od.offerings_description), '') IS NOT NULL) AS offerings_description,
    string_agg(DISTINCT NULLIF(TRIM(BOTH FROM tc.top_customers), ''), ' | ')

    b.innovation_roadmap,
    b.future_projections,
    b.market_share_percentage,
    b.tam,
    b.sam,
    b.som,
    b.customer_concentration_risk
  FROM company_business b
  LEFT JOIN company_focus_sectors fs
    ON fs.company_business_id = b.company_business_id
  LEFT JOIN company_offerings_description od
    ON od.company_business_id = b.company_business_id
  LEFT JOIN company_top_customers tc
    ON tc.company_business_id = b.company_business_id
  LEFT JOIN company_key_competitors kc
    ON kc.company_business_id = b.company_business_id

    b.company_id,

    company_skills_view.company_id,
    jsonb_agg(
      jsonb_build_object(
        'skill_set_id', company_skills_view.skill_set_id,
        'skill_name', company_skills_view.skill_set_name,
        'skill_short_name', company_skills_view.skill_short_name,
        'required_level', company_skills_view.required_level,
        'proficiency_name', company_skills_view.proficiency_name,
        'proficiency_code', company_skills_view.proficiency_code
      )
      ORDER BY company_skills_view.required_level DESC, company_skills_view.skill_set_name
